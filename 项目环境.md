# 前端上传服务器流程
# 1.本地构建
cocos create构建发布
# 2.上传本地文件到服务器
scp -r D:\CocosProjects\NsyQQQ\build\wechatgame nsy@8.135.50.241:/tmp/doudizhu/
# 3.拷贝到 Nginx 静态目录
sudo cp -r /tmp/doudizhu/wechatgame/* /var/www/game/
# 4.测试并重载 Nginx 
sudo nginx -t && sudo nginx -s reload

# 后端上传服务器流程
# 1.本地构建
npm run build
# 2.上传本地文件到服务器
scp -r D:\CocosProjects\NsyQQQ\server\dist nsy@8.135.50.241:/tmp/doudizhu/
# 3.拷贝进 Docker 容器
sudo docker cp dist doudizhu:/app/
# 4.重启 Docker 容器
sudo docker restart doudizhu

# 查看服务器日志
sudo docker logs -f doudizhu
# 查看 Nginx 配置
sudo cat /etc/nginx/nginx.conf

# 启动本地服务器
npm run dev
